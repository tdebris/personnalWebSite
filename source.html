Symbol Sampling with Probability Distribution in Python

import random

# Define symbols and their respective probabilities
symbols = [' ', 'i', 'r', 'a', 'j', 's', 'b', 'k', 't', 'c', 'l', 'u', 
           'd', 'm', 'v', 'e', 'n', 'w', 'f', 'o', 'x', 'g', 'p', 'y', 
           'h', 'q', 'z']
probabilities = [0.1834, 0.0591, 0.0555, 0.0640, 0.0023, 0.0697, 0.0064, 
                 0.0001, 0.0572, 0.0259, 0.0465, 0.0506, 0.0260, 0.0245, 
                 0.0100, 0.1486, 0.0623, 0.0001, 0.0078, 0.0459, 0.0031, 
                 0.0083, 0.0256, 0.0021, 0.0061, 0.0081, 0.0008]

# Step 1: Normalize probabilities in case they don't sum exactly to 1
probabilities = [p / sum(probabilities) for p in probabilities]

# Step 2: Create a cumulative distribution for sampling
cumulative_probs = []
cumulative = 0
for p in probabilities:
    cumulative += p
    cumulative_probs.append(cumulative)

# Function to sample symbols based on the cumulative distribution
def sample_from_distribution():
    """Sample a symbol based on the defined probability distribution."""
    r = random.random()  # Generate a random float between 0 and 1
    for i, cp in enumerate(cumulative_probs):
        if r < cp:
            return symbols[i]

# Example usage: Generate a list of 10 sampled symbols
samples = [sample_from_distribution() for _ in range(10)]
print("Sampled symbols:", samples)
