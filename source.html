Huffman Code

This program demonstrates sampling based on symbol probabilities using SageMath.


# SageMath code: Huffman Coding with Symbol Sampling

import random

# Define symbols and their probabilities
symbols = [' ', 'i', 'r', 'a', 'j', 's', 'b', 'k', 't', 'c', 'l', 'u', 
           'd', 'm', 'v', 'e', 'n', 'w', 'f', 'o', 'x', 'g', 'p', 'y', 
           'h', 'q', 'z']
probabilities = [0.1834, 0.0591, 0.0555, 0.0640, 0.0023, 0.0697, 0.0064, 
                 0.0001, 0.0572, 0.0259, 0.0465, 0.0506, 0.0260, 0.0245, 
                 0.0100, 0.1486, 0.0623, 0.0001, 0.0078, 0.0459, 0.0031, 
                 0.0083, 0.0256, 0.0021, 0.0061, 0.0081, 0.0008]

# Normalize probabilities
probabilities = [p / sum(probabilities) for p in probabilities]

# Create cumulative distribution for sampling
cumulative_probs = []
cumulative = 0
for p in probabilities:
    cumulative += p
    cumulative_probs.append(cumulative)

# Sample from the distribution
def sample_from_distribution():
    r = random.random()
    for i, cp in enumerate(cumulative_probs):
        if r < cp:
            return symbols[i]

# Example usage
samples = [sample_from_distribution() for _ in range(10)]
print("Sampled symbols:", samples)
